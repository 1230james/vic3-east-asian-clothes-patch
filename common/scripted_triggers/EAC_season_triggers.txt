EAC_character_cool_trigger = {
	exists = scope:character
	scope:character = {
        month >= 3 # month compares 0 - 11
        month <  9
	}
}

EAC_pop_cool_trigger = {
	exists = scope:pop
	scope:pop = {
		month >= 3
        month <  9
	}
}
